Tier-Checking Strategy
The tier system would control access to features based on the organization's subscription level. Here's how it would work:

1. Database Schema Addition
You'd add a tier field to track the organization's subscription level and feature entitlements.

2. Middleware & Access Control
Similar to how the app currently uses role-based access control (like requireAdmin, requireSuperAdmin), you'd implement tier-based middleware:

// Example tier middleware
const requireTier = (minTier: 'basic' | 'pro' | 'premium') => {
  return async (req, res, next) => {
    const org = await getOrganization(req.user.orgId);
    
    if (!canAccessFeature(org.tier, minTier)) {
      return res.status(403).json({ 
        message: "Upgrade required",
        currentTier: org.tier,
        requiredTier: minTier 
      });
    }
    next();
  };
};

import { useOrganizationTier } from '@/hooks/useOrganizationTier';
function AdminDashboard() {
  const { tier, canAccess } = useOrganizationTier();
  
  return (
    <>
      {/* Always available */}
      <Link href="/admin/content">Manage Content</Link>
      
      {/* Pro+ only */}
      {canAccess('pro') && (
        <Link href="/admin/ab-testing">A/B Testing</Link>
      )}
      
      {/* Premium only */}
      {canAccess('premium') && (
        <>
          <Link href="/admin/automation-rules">Automation</Link>

        </>
      )}
      
      {/* Upgrade prompts */}
      {!canAccess('premium') && (

      )}
    </>
  );
}
